def verificar_guias_vencidas():
    """Alerta por guías pendientes sin procesar — SAP Delivery Due List."""
    from datetime import date, timedelta
    
    guias_vencidas = db.query("""
        SELECT numero_guia, tipo_guia, fecha_esperada, proveedor_origen
        FROM guias
        WHERE estado IN ('PENDIENTE', 'EN_PROCESO')
        AND fecha_esperada < :hoy
    """, {'hoy': date.today().isoformat()})
    
    for guia in guias_vencidas:
        dias_retraso = (date.today() - date.fromisoformat(guia['fecha_esperada'])).days
        disparar_alerta(
            tipo='GUIA_VENCIDA',
            mensaje=f"Guía {guia['numero_guia']} con {dias_retraso} día(s) de retraso. "
                    f"Proveedor: {guia['proveedor_origen']}",
            modulo='GUIAS'
        )

def verificar_diferencias_recepcion():
    """Alerta por diferencias significativas en recepciones del día."""
    diferencias = db.query("""
        SELECT guia_id, COUNT(*) as items_con_diferencia,
               SUM(ABS(diferencia)) as total_diferencia
        FROM diferencias_recepcion
        WHERE DATE(timestamp) = DATE('now')
        GROUP BY guia_id
        HAVING total_diferencia > 10
    """)
    
    for d in diferencias:
        disparar_alerta(
            tipo='DIFERENCIA_RECEPCION',
            mensaje=f"Guía {d['guia_id']}: {d['items_con_diferencia']} ítems con diferencia "
                    f"({d['total_diferencia']} unidades total)",
            modulo='RECEPCION'
        )
